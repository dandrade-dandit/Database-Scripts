SELECT      CAD.EMP_NUMERO_MATRICULA
,           CAD.EMP_DATA_NASCIMENTO
,           CAR_CONF.FUN_CODIGO AS EMP_FUN_CODIGO
,           CAD.EMP_DATA_ADMISSAO
,           CAR.CAR_CODIGO_NIVEL AS EMP_CAR_CODIGO_NIVEL
,           CAD.EMP_NOME
,           CAD.EMP_NOME_ABREVIADO
,           CAD.EMP_NUMERO_CELULAR
,           CAD.EMP_DDD_CELULAR
,           CAD.EMP_OEX_CODIGO_CEDIDO
,           CAD.EMP_OEX_CODIGO_REQUISITADO
,           CAR.CAR_NOME AS EMP_QLP_CARGO
,           CAR.CAR_SIGLA AS EMP_QLP_SIGLA
,           DEP2.DEP_TIPO AS EMP_DEP_FISICO_TIPO
,           DECODE(DEP2.DEP_TIPO, 1,'SEDE', 2,'SUPERINTENDÊNCIA', 3,'AEROPORTO', 4,'GTA',5,'COORDENAÇÃO DE NEGÓCIO', 6,'GNA',7,'UNA', 8,'CONVÊNIOS COM RESSARCIMENTO', 9,'CONVÊNIOS SEM RESSARCIMENTO',10,'DEPENDÊNCIA GNA', 11,'DEPENDÊNCIA UNA',12,'DEPENDÊNCIA DE DEPENDÊNCIA', DEP2.DEP_TIPO) AS EMP_DEP_FISICO_TIPO_NOME
,           DECODE(CAD.EMP_QLP_CAR_CODIGO_NIVEL,1,'OPERACIONAL',2,'TECNICO',3,'ESPECIALIZADO',4,'PLENO',5,'SENIOR','') AS EMP_QLP_CAR_NIVEL
,           OCC.OCC_DESCRICAO AS EMP_QLP_CAR_OCUPACAO
,           UOR.UOR_SIGLA AS EMP_UOR_SIGLA
,           UOR.UOR_NOME AS EMP_UOR_NOME
,           (SELECT UOR_NOME FROM UNIDADES_ORGANIZACIONAIS WHERE UOR_CODIGO = CAD.EMP_QFU_HFL_UOR_COD_SUBST) AS EMP_UOR_SIGLA_NOME
,           UOR2.UOR_SIGLA AS EMP_UOR2_SIGLA
,           UOR2.UOR_NOME AS EMP_UOR2_LOTACAO
,           UOR3.UOR_SIGLA AS EMP_UOR3_SIGLA
,           UOR3.UOR_NOME AS EMP_UOR3_LOTACAO
,           UOR4.UOR_SIGLA AS EMP_UOR4_SIGLA
,           UOR4.UOR_NOME AS EMP_UOR4_LOTACAO
,           UOR6.UOR_SIGLA AS UOR_SIGLA_FISICO
,           UOR6.UOR_NOME AS UOR_NOME_FISICO
,           CAD.EMP_UOR_CODIGO_FISICO AS UOR_CODIGO_FISICO
,           CAR_CONF.FUN_NOME AS EMP_QFU_FUNCAO
,           CAD.EMP_QFU_FUN_CODIGO
,           CAD.EMP_QFU_FUN_CODIGO_ACUMULA
,           CAD.EMP_QFU_HFL_UOR_COD_ACUM
,           CAD.EMP_QFU_FUN_CODIGO_SUBSTITUI
,           (SELECT FUN_NOME FROM CARGOS_CONFIANCA WHERE FUN_CODIGO = CAD.EMP_QFU_FUN_CODIGO_SUBSTITUI) AS EMP_QFU_FUN_DSC_SUBSTITUI
,           CAD.EMP_QFU_HFL_UOR_COD_SUBST
,           DEP.DEP_NOME AS EMP_DEP_LOTACAO
,           CAD.EMP_DEP_CODIGO_LOTACAO
,           DEP.DEP_SIGLA AS EMP_DEP_LOTACAO_SIGLA
,           DEP2.DEP_NOME AS EMP_DEP_FISICO
,           CAD.EMP_DEP_CODIGO_FISICO
,           DEP2.DEP_SIGLA AS EMP_DEP_FISICO_SIGLA
,           UOR.UOR_NOME AS EMP_UOR_LOTACAO
,           DEP.EMP_UOR_CODIGO_LOTACAO
,           DECODE(CAD.EMP_STATUS,0,'TEMPORARIAMENTE AFASTADO',1,'ATIVO',2,'DESLIGADO',CAD.EMP_STATUS) AS EMP_STATUS
,           CAD.EMP_NOME_GUERRA
,           CAD.EMP_ENDERECO_ELETRONICO_MAIL
,           ( SELECT F.EFC_UFF_VOIP_DDD FROM EMPREGADO_FONE_COMERCIAL F WHERE CAD.EMP_NUMERO_MATRICULA = F.EFC_EMP_NUMERO_MATRICULA AND F.EFC_UFF_TIPO= 1 AND ROWNUM < 2) AS EMP_EFC_UFF_VOIP_DDD_DIRETO 
,           ( SELECT F.EFC_UFF_VOIP_DDD FROM EMPREGADO_FONE_COMERCIAL F WHERE CAD.EMP_NUMERO_MATRICULA = F.EFC_EMP_NUMERO_MATRICULA AND F.EFC_UFF_TIPO= 2 AND ROWNUM < 2) AS EMP_EFC_UFF_VOIP_DDD_RAMAL 
,           ( SELECT F.EFC_UFF_VOIP_DDD FROM EMPREGADO_FONE_COMERCIAL F WHERE CAD.EMP_NUMERO_MATRICULA = F.EFC_EMP_NUMERO_MATRICULA AND F.EFC_UFF_TIPO= 3 AND ROWNUM < 2) AS EMP_EFC_UFF_VOIP_DDD_FAX 
,           ( SELECT F.EFC_UFF_VOIP_DDD FROM EMPREGADO_FONE_COMERCIAL F WHERE CAD.EMP_NUMERO_MATRICULA = F.EFC_EMP_NUMERO_MATRICULA AND F.EFC_UFF_TIPO= 4 AND ROWNUM < 2) AS EMP_EFC_UFF_VOIP_DDD  -- EXIBE CAMPO DDD VOIP
,           ( SELECT F.EFC_UFF_NUMERO FROM EMPREGADO_FONE_COMERCIAL F WHERE CAD.EMP_NUMERO_MATRICULA = F.EFC_EMP_NUMERO_MATRICULA AND F.EFC_UFF_TIPO= 1 AND ROWNUM < 2) AS EMP_EFC_UFF_FONE_DIRETO 
,           ( SELECT F.EFC_UFF_NUMERO FROM EMPREGADO_FONE_COMERCIAL F WHERE CAD.EMP_NUMERO_MATRICULA = F.EFC_EMP_NUMERO_MATRICULA AND F.EFC_UFF_TIPO= 2 AND ROWNUM < 2) AS EMP_EFC_UFF_RAMAL 
,           ( SELECT F.EFC_UFF_NUMERO FROM EMPREGADO_FONE_COMERCIAL F WHERE CAD.EMP_NUMERO_MATRICULA = F.EFC_EMP_NUMERO_MATRICULA AND F.EFC_UFF_TIPO= 3 AND ROWNUM < 2) AS EMP_EFC_UFF_FAX
,           ( SELECT F.EFC_UFF_NUMERO FROM EMPREGADO_FONE_COMERCIAL F WHERE CAD.EMP_NUMERO_MATRICULA = F.EFC_EMP_NUMERO_MATRICULA AND F.EFC_UFF_TIPO= 4 AND ROWNUM < 2) AS EMP_EFC_UFF_VOIP -- EXIBE CAMPO NUMERO VOIP 
,           HA.ORDEM AS EMP_ORDEM 
,           CAD.EMP_UOR_CODIGO_FISICO_AGENDA
,           UOR5.UOR_SIGLA EMP_UOR_SIGLA_FISICO_AGENDA
,           UOR5.UOR_NOME EMP_UOR_FISICO_AGENDA
,           ( SELECT  B.UOR_VIRTUAL_CODIGO FROM  EMPREGADO_UORG_VIRTUAL A, UNIDADE_ORGANIZACIONAL_VIRTUAL B WHERE A.UOR_VIRTUAL_CODIGO = B.UOR_VIRTUAL_CODIGO AND A.EMP_NUMERO_MATRICULA =  CAD.EMP_NUMERO_MATRICULA) AS UOR_CODIGO_VIRTUAL
,           ( SELECT  B.UOR_VIRTUAL_SIGLA  FROM  EMPREGADO_UORG_VIRTUAL A, UNIDADE_ORGANIZACIONAL_VIRTUAL B WHERE A.UOR_VIRTUAL_CODIGO = B.UOR_VIRTUAL_CODIGO AND A.EMP_NUMERO_MATRICULA =  CAD.EMP_NUMERO_MATRICULA) AS UOR_SIGLA_VIRTUAL
,           DECODE(( SELECT  B.UOR_VIRTUAL_SIGLA||' - '|| B.UOR_VIRTUAL_NOME FROM  EMPREGADO_UORG_VIRTUAL A, UNIDADE_ORGANIZACIONAL_VIRTUAL B WHERE A.UOR_VIRTUAL_CODIGO = B.UOR_VIRTUAL_CODIGO AND A.EMP_NUMERO_MATRICULA =  CAD.EMP_NUMERO_MATRICULA),'','', ( SELECT  B.UOR_VIRTUAL_SIGLA||' - '|| B.UOR_VIRTUAL_NOME FROM  EMPREGADO_UORG_VIRTUAL A, UNIDADE_ORGANIZACIONAL_VIRTUAL B WHERE A.UOR_VIRTUAL_CODIGO = B.UOR_VIRTUAL_CODIGO AND   A.EMP_NUMERO_MATRICULA =  CAD.EMP_NUMERO_MATRICULA)||' ('||UOR5.UOR_SIGLA||')') AS UOR_DSC_VIRTUAL
,           DECODE(( SELECT  B.UOR_VIRTUAL_CODIGO FROM  EMPREGADO_UORG_VIRTUAL A, UNIDADE_ORGANIZACIONAL_VIRTUAL B WHERE A.UOR_VIRTUAL_CODIGO = B.UOR_VIRTUAL_CODIGO AND A.EMP_NUMERO_MATRICULA =  CAD.EMP_NUMERO_MATRICULA),NULL,'O','V') AS UOR_FLAG_VIRTUAL
,           CAD.EMP_NUMERO_CPF
,           CAD.EMP_NSA_CODIGO_NIVEL
,           CAD.EMP_NSA_CODIGO_PADRAO
,           DECODE(CAD.EMP_QFU_FUN_CODIGO, NULL, NSA.NSA_VLD_CODIGO, CFU.CFU_VLD_CODIGO) VLD_CODIGO
,           CAR_CONF.FUN_CODIGO COD_FUNCAO_CONFIANCA
,           CAR_CONF.FUN_NOME FUNCAO_CONFIANCA
,           CAD.EMP_NU_CARTEIRA_IDENTIDADE RG
,           TO_NUMBER(CC.CD_CENTRO_CUSTO||CC.DV_CENTRO_CUSTO) COD_CENTRO_CUSTO
,           CC.DS_CENTRO_CUSTO CENTRO_CUSTO
,           CFU.CFU_CODIGO COD_CATEGORIA_FUNCAO
,           CFU.CFU_NOME CATEGORIA_FUNCAO
,           CAD.EMP_SFU_CODIGO
FROM        CADASTROS CAD
,           CARGOS CAR
,           OCUPACOES_CARGO OCC
,           DEPENDENCIAS DEP
,           DEPENDENCIAS DEP2
,           UNIDADES_ORGANIZACIONAIS UOR
,           UNIDADES_ORGANIZACIONAIS UOR2
,           UNIDADES_ORGANIZACIONAIS UOR3
,           UNIDADES_ORGANIZACIONAIS UOR4
,           UNIDADES_ORGANIZACIONAIS UOR5
,           UNIDADES_ORGANIZACIONAIS UOR6
,           QUADRO_FUNCOES QF
,           CARGOS_CONFIANCA CAR_CONF
,           HIERARQUIA_AGENDA HA
,           NIVEIS_SALARIAIS NSA
,           CATEGORIAS_FUNCOES CFU
,           CENTRO_CUSTO CC
WHERE       CAD.EMP_STATUS <> 2 
AND         UOR.UOR_TIPO_UNIDAD_ORG <> 14 
AND         NSA.NSA_CODIGO_NIVEL (+)= CAD.EMP_NSA_CODIGO_NIVEL  
AND         NSA.NSA_CODIGO_PADRAO (+)= CAD.EMP_NSA_CODIGO_PADRAO  
AND         NSA.NSA_INDICADOR_NIVEL_ESCOLA (+)= CAD.EMP_NSA_IN_NIVEL_ESCOLARIDADE  
AND         CFU.CFU_CODIGO (+)= CAR_CONF.FUN_CFU_CODIGO  
AND         CAR.CAR_CODIGO (+)= CAD.EMP_QLP_CAR_CODIGO 
AND         CAR.CAR_CODIGO_NIVEL (+)= CAD.EMP_QLP_CAR_CODIGO_NIVEL 
AND         CAR.CAR_OCC_CODIGO (+)= CAD.EMP_QLP_CAR_OCC_CODIGO 
AND         OCC.OCC_CODIGO (+)=CAR.CAR_OCC_CODIGO 
AND         QF.QFU_FUN_CODIGO (+)= CAD.EMP_QFU_FUN_CODIGO 
AND         QF.QFU_HFL_UOR_CODIGO (+)= CAD.EMP_QFU_HFL_UOR_CODIGO
AND         QF.QFU_HFL_DATA_VIGENCIA (+)= CAD.EMP_QFU_HFL_DATA_VIGENCIA 
AND         TO_NUMBER(CC.CD_CENTRO_CUSTO||CC.DV_CENTRO_CUSTO) = CAD.EMP_ATI_CT_CUSTOS (+) 
AND         CAR_CONF.FUN_CODIGO (+)= CAD.EMP_QFU_FUN_CODIGO 
AND         DEP.DEP_CODIGO (+)= CAD.EMP_DEP_CODIGO_LOTACAO 
AND         DEP2.DEP_CODIGO (+)= CAD.EMP_DEP_CODIGO_FISICO 
AND         UOR.UOR_CODIGO(+)= CAD.EMP_UOR_CODIGO_LOTACAO 
AND         UOR2.UOR_CODIGO (+)= UOR.UOR_UOR_CODIGO 
AND         UOR3.UOR_CODIGO (+)= UOR2.UOR_UOR_CODIGO 
AND         UOR4.UOR_CODIGO (+)=UOR3.UOR_UOR_CODIGO 
AND         UOR5.UOR_CODIGO (+)= CAD.EMP_UOR_CODIGO_FISICO_AGENDA 
AND         UOR6.UOR_CODIGO (+)= CAD.EMP_UOR_CODIGO_FISICO 
AND         HA.EMP_NUMERO_MATRICULA (+)= CAD.EMP_NUMERO_MATRICULA
